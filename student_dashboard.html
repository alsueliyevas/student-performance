# ── Student Performance — Python EDA ──────────────────
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# 1. Yüklə və hazırla
df = pd.read_excel("performancepivot.xlsx", sheet_name="StudentsPerformance")
df.columns = ['gender','race','parental_edu','lunch',
              'test_prep','math','reading','writing']
df['average'] = (df['math'] + df['reading'] + df['writing']) / 3
df['grade'] = pd.cut(df['average'], bins=[0,60,70,80,90,100],
                      labels=['F','D','C','B','A'])

# 2. EDA nəticələri (1000 sətir, 0 null)
print(df[['math','reading','writing']].describe().round(2))

# 3. Korrelyasiya (oxuma×yazı = 0.95!)
corr = df[['math','reading','writing']].corr()

# 4. Cinslər üzrə müqayisə
gender_stats = df.groupby('gender')[['math','reading','writing']].mean()
# female: math=63.6, reading=72.6, writing=72.5
# male:   math=68.7, reading=65.5, writing=63.3

# 5. Test hazırlığı təsiri
prep_impact = df.groupby('test_prep')['average'].mean()
# completed: 72.67 (+7.6 vs none)

# 6. Valideyn təhsili — ən güclü predictor
edu_order = ['some high school','high school',
             "associate's degree",'some college',
             "bachelor's degree","master's degree"]
edu_avg = df.groupby('parental_edu')['average'].mean().reindex(edu_order)
# 63.1 → 73.6 (aşağıdan yuxarı artım)

# 7. Vizuallaşdırma — Seaborn boxplot
fig, axes = plt.subplots(2, 3, figsize=(18,10))
sns.boxplot(data=df, x='race', y='math',
            palette='husl', ax=axes[0,0])
sns.violinplot(data=df, x='gender', y='average',
               palette=['#6c63ff','#ff6584'], ax=axes[0,1])
sns.heatmap(corr, annot=True, cmap='RdYlGn', ax=axes[0,2])
plt.tight_layout()
plt.savefig('analysis.png', dpi=150)

# 8. Kecid faizi
for col in ['math','reading','writing']:
    pct = (df[col] >= 60).mean() * 100
    print(f"{col}: {pct:.1f}%")
# math: 67.7%, reading: 74.3%, writing: 71.9%
